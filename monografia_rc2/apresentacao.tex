  \section{Métricas}
  \begin{frame}
   \frametitle{Distribuição de Grau}
   \begin{columns}[c]
    \column{.5\textwidth}
     \begin{figure}
      \includegraphics[width=\textwidth]{degree.pdf}
      \caption{Distribuição de grau da rede 15m\cite{borge2011structural}}
     \end{figure}
    \column{.5\textwidth}
     \begin{itemize}
      \item Distribuição de grau evidencia como as conexões se distribuem entre os vértices.
      \item Constante de tempo: $\tau \sim \langle k \rangle$.
      \item Limiar de propagação: $\lambda \sim 1/\langle k^2 \rangle$.
      \item Redes livre de escala possuem $\langle k^2 \rangle \to \infty$, logo $\lambda \to 0$.
     \end{itemize}
   \end{columns}
  \end{frame}
  
  \begin{frame}
   \frametitle{Distribuição de \textit{clustering}}
   \begin{columns}[c]
    \column{.5\textwidth}
     \begin{figure}
      \includegraphics[width=\textwidth]{clustering.pdf}
      \caption{Distribuição de \textit{clustering} da rede 15m\cite{borge2011structural}}
     \end{figure}
    \column{.5\textwidth}
     \begin{itemize}
      \item Medidas de aglomeração medem a ocorrência de laços pequenos na rede.
      \item Em geral, $C_i \sim k_i^{-1}$.
      \item Constante de tempo: $\tau \sim 1/C_i$.
      \item Número reprodutivo: $R_0 \sim 1/C_i$.
     \end{itemize}
   \end{columns}
  \end{frame}
  
  \begin{frame}
   \frametitle{Assortatividade}
   \begin{columns}[c]
    \column{.5\textwidth}
     \begin{figure}
      \includegraphics[width=\textwidth]{correlation.pdf}
      \caption{Correlação de grau da rede 15m\cite{borge2011structural}}
     \end{figure}
    \column{.5\textwidth}
     \begin{itemize}
      \item Correlação de grau $r$ informa o quanto vértices se conectam a outros similares.
      \item Redes assortativas são resistentes à remoção de \textit{hubs}.
      \item Disseminações com estado removido devem possuir maior cobertura em redes assortativas.
     \end{itemize}
   \end{columns}
  \end{frame}
  
  \begin{frame}
   \frametitle{\textit{$k$-core}}
   \begin{columns}[c]
    \column{.4\textwidth}
     \begin{figure}
      \includegraphics[width=\textwidth]{k-core.pdf}
      \caption{Distribuição de \textit{coreness} da rede 15m\cite{borge2011structural}}
     \end{figure}
    \column{.6\textwidth}
     \begin{itemize}
      \item Redes se organizam em camadas, onde o núcleo interno está altamente interconectado.
      \item Pode-se obter o \textit{$k$-core} removendo repetidamente todos os vértices com grau menor que $k$.
      \item O \textit{coreness} de um vértice é o núcleo com maior $k$ ao qual ele pertence.
      \item Disseminações que se iniciam no núcleo mais interno possuem maior sucesso\cite{kitsak2010identification}.
     \end{itemize}
   \end{columns}
  \end{frame}
  
  \section{Modelos de rede}
  
  \begin{frame}
   \frametitle{Modelo de Erdös-Renyi}
   \begin{columns}
    \column{0.5\textwidth}
    \begin{figure}
     \includegraphics[width=\textwidth]{er.pdf}
    \end{figure}
    
    \column{0.5\textwidth}
    Parâmetros:
    \begin{itemize}
     \item $n$: número de vértices
     \item $k$: grau médio do grafo     
    \end{itemize}
    Para cada par de vértices, cria uma aresta com probabilidade $p = k/n$.
    \begin{itemize}
     \item Mundo pequeno: $d \sim \log n$;
     \item Transitividade é igual a $p$;
     \item Distribuição de graus: Poisson.
     \item Modelo muito simples, também chamado simplesmente ``aleatório''
    \end{itemize}
   \end{columns}

  \end{frame}

  \begin{frame}
   \frametitle{Modelo de Watts-Strogatz (1998)}
   \begin{columns}
    \column{0.4\textwidth}
    \begin{figure}
     \includegraphics[width=\textwidth]{ws.pdf}
    \end{figure}
    
    \column{0.6\textwidth}
    Parâmetros:
    \begin{itemize}
     \item $n$: número de vértices
     \item $k$: número de vizinhos aos quais cada vértice se conecta
     \item $\beta$: probabilidade de reconectar um extremo de cada aresta a outro vértice aleatório
    \end{itemize}
    Este modelo vai de um grafo regular ($\beta = 0$) a um aleatório ($\beta = 1$).
    \begin{itemize}
     \item Mundo pequeno: $d \sim \log n$;
     \item Transitividade entre $\frac{3}{4} \frac{k-2}{k-1} \leq C \leq \frac{k}{n}$;
     \item Distribuição de graus: Poisson;
     \item Modelo inclui fenômeno de mundo pequeno e alta transitividade.
    \end{itemize}
   \end{columns}

  \end{frame}
  
  \begin{frame}
   \frametitle{Modelo de Barabási-Albert}
   \begin{columns}
    \column{0.4\textwidth}
    \begin{figure}
     \includegraphics[width=\textwidth]{ba.pdf}
    \end{figure}
    
    \column{0.6\textwidth}
    Parâmetros:
    \begin{itemize}
     \item $n$: número de vértices
     \item $k$: número de arestas que cada novo vértice adiciona à rede.
    \end{itemize}
    Também chamado de \textit{rich-get-richer}, cada novo vértice se conecta, preferencialmente, aos vértices mais conectados.
    \begin{itemize}
     \item Mundo ultra-pequeno: $d \sim \log n / \log \log n$;
     \item Transitividade depende do tamanho da rede $C \sim n^{-3/4}$;
     \item Distribuição de graus: Pareto;
     \item Mecanismo de \textit{ligação preferencial} encontra paralelos em múltiplos sistemas.
    \end{itemize}
   \end{columns}
  \end{frame}
  
  \begin{frame}
   \frametitle{Modelo de Ravasz-Barabási}
   \begin{columns}
    \column{0.4\textwidth}
    \begin{figure}
     \includegraphics[width=\textwidth]{t3.pdf}
    \end{figure}
    
    \column{0.6\textwidth}
    Parâmetros:
    \begin{itemize}
     \item $l$: número de níveis de recursão
     \item $k$: número de vértices por subgrafo completo.
     \item $n$: $k^{l+1}$
    \end{itemize}
    Nós periféricos do nível $L-1$ se conectam ao nó central do nível $L$.
    \begin{itemize}
     \item Mundo pequeno: $d \sim l \sim \log n$;
     \item Transitividade inversamente proporcional ao grau do vértice: $C \sim k^{-1}$;
     \item Distribuição de graus segue lei de potência;
     \item Primeira inclusão dos três principais fenômenos em um modelo.
    \end{itemize}
   \end{columns}
  \end{frame}
